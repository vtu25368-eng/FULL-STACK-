// State variables

let isLoggedIn = false;

let projects = []; // Array of project objects: { id, title, description, techStack, githubLink }

let currentProjectId = null; // For detail page

 

// DOM elements

const pages = {

    landing: document.getElementById('landing'),

    login: document.getElementById('login'),

    upload: document.getElementById('upload'),

    listing: document.getElementById('listing'),

    detail: document.getElementById('detail'),

    dashboard: document.getElementById('dashboard')

};

 

// Utility: Show a specific page

function showPage(pageId) {

    Object.values(pages).forEach(page => page.style.display = 'none');

    pages[pageId].style.display = 'block';

}

 

// Utility: Render project cards

function renderProjectCard(project, container, showDelete = false) {

    const card = document.createElement('div');

    card.className = 'col-md-4 mb-3';

    card.innerHTML = `

        <div class="card">

            <img src="https://via.placeholder.com/300x200?text=Project" alt="Project" class="card-img-top">

            <div class="card-body">

                <h5 class="card-title">${project.title}</h5>

                <p class="card-text">${project.description.substring(0, 100)}...</p>

                ${showDelete ? `<button class="btn btn-danger mt-2 delete-btn" data-id="${project.id}">Delete</button>` : ''}

            </div>

        </div>

    `;

    card.addEventListener('click', () => showDetail(project.id));

    container.appendChild(card);

}

 

// Utility: Render empty state

function renderEmptyState(container) {

    container.innerHTML = `

        <div class="empty-state">

            <img src="https://via.placeholder.com/200?text=No+Projects" alt="Empty">

            <p>No projects yet. Add some!</p>

        </div>

    `;

}

 

// Show detail page

function showDetail(id) {

    const project = projects.find(p => p.id === id);

    if (!project) return;

    document.getElementById('detailContent').innerHTML = `

        <img src="https://via.placeholder.com/400x200?text=Project+Image" alt="Project" class="img-fluid mb-3">

        <h2>${project.title}</h2>

        <p>${project.description}</p>

        <p><strong>Tech Stack:</strong> ${project.techStack}</p>

        <a href="${project.githubLink}" class="btn btn-primary" target="_blank">View on GitHub</a>

    `;

    showPage('detail');

}

 

// Event listeners

document.getElementById('loginBtn').addEventListener('click', () => showPage('login'));

document.getElementById('viewProjectsBtn').addEventListener('click', () => showPage('listing'));

document.getElementById('addProjectBtn').addEventListener('click', () => {

    if (!isLoggedIn) {

        showPage('login');

        return;

    }

    showPage('upload');

});

 

// Login form

document.getElementById('loginForm').addEventListener('submit', (e) => {

    e.preventDefault();

    isLoggedIn = true;

    showPage('dashboard');

});

 

// Upload form

document.getElementById('uploadForm').addEventListener('submit', (e) => {

    e.preventDefault();

    if (!isLoggedIn) {

        showPage('login');

        return;

    }

    const title = document.getElementById('title').value;

    const description = document.getElementById('description').value;

    const techStack = document.getElementById('techStack').value;

    const githubLink = document.getElementById('githubLink').value;

    projects.push({ id: Date.now(), title, description, techStack, githubLink });

    document.getElementById('uploadForm').reset();

    showPage('dashboard');

});

 

// Initial render and updates

function updateListing() {

    const container = document.getElementById('projectList');

    container.innerHTML = '';

    if (projects.length === 0) {

        renderEmptyState(container);

    } else {

        projects.forEach(project => renderProjectCard(project, container));

    }

}

 

function updateDashboard() {

    const container = document.getElementById('dashboardList');

    container.innerHTML = '';

    if (projects.length === 0) {

        renderEmptyState(container);

    } else {

        projects.forEach(project => renderProjectCard(project, container, true));

    }

    // Add delete listeners

    container.querySelectorAll('.delete-btn').forEach(btn => {

        btn.addEventListener('click', (e) => {

            e.stopPropagation();

            const id = parseInt(e.target.dataset.id);

            projects = projects.filter(p => p.id !== id);

            updateDashboard();

        });

    });

}

 

// Hash-based navigation (simple routing)

window.addEventListener('hashchange', () => {

    const hash = window.location.hash.substring(1);

    if (pages[hash]) {

        if (hash === 'upload' || hash === 'dashboard') {

            if (!isLoggedIn) {

                showPage('login');

                return;

            }

        }

        showPage(hash);

        if (hash === 'listing') updateListing();

        if (hash === 'dashboard') updateDashboard();

    } else {

        showPage('landing');

    }

});

 

// Initial load

window.addEventListener('load', () => {

    showPage('landing');

});
